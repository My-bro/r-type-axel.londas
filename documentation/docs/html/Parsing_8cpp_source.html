52void Parsing::getArgs(char **av)
53{
54 std::size_t cpt = 1;
55 std::string current_argument = std::string(av[0]);
56 std::string next_argument;
57
58 if (current_argument == "r-type_server") {

85void Parsing::getFileInfos(std::string& filename)
86{
87 std::ifstream file(filename);
88 if (!file.is_open()) {
89 throw std::exception();
90 }
91
92 std::string line;
93 std::string currentSection;
94
95 while (std::getline(file, line)) {
96 line.erase(0, line.find_first_not_of(" \t"));
97 line.erase(line.find_last_not_of(" \t") + 1);
98 if (line.empty() || line[0] == ';' || line[0] == '#') {
99 continue;
100 }
101 if (line.front() == '[' && line.back() == ']') {
102 currentSection = line.substr(1, line.size() - 2);
103 } else {
104 auto delimiterPos = line.find('=');
105 if (delimiterPos != std::string::npos) {
106 size_t commentPos = line.find_first_of(" \t\n;#");
107 if (commentPos != std::string::npos) {
108 line = line.substr(0, commentPos);
109 }
110 std::string key = line.substr(0, delimiterPos);
111 std::string value = line.substr(delimiterPos + 1);
112 key.erase(0, key.find_first_not_of(" \t"));
113 key.erase(key.find_last_not_of(" \t") + 1);
114 value.erase(0, value.find_first_not_of(" \t"));
115 value.erase(value.find_last_not_of(" \t") + 1);
116 if (currentSection == "server") {
117 serverSettings[key] = value;
118 } else if (currentSection == "Client") {
119 clientSettings[key] = value;
120 }
121 }
122 }
123 }
